    \textbf{\Large{交比（Cross Ratio）}}
\end{center}

\indent 交比是射影幾何的一個重要概念，表面上是表示四個共線點的相對位置關係，實際是蘊含着射影幾何的不變性質。
在競賽幾何題中，交比是線段變換的重要工具，也是三點共線和三線共點的橋樑。

\section{線交比}
\begin{mybox}[定義]{4em}{1em}{0.5em}
    對於同一直線上的四個相異點$A$、$B$、$P$和$Q$，定義
    $$
        \left(A,B;P,Q\right)=\frac{\overrightarrow{AP}/\overrightarrow{PB}}{\overrightarrow{AQ}/\overrightarrow{QB}}
    $$
    和
    $$
        \left[A,B;P,Q\right]=\frac{AP/PB}{AQ/QB}
    $$
    稱$\left(A,B;P,Q\right)$為線段$AB$關於分點$P$和$Q$的\underline{有向交比}；

    稱$\left[A,B;P,Q\right]$為線段$AB$關於分點$P$和$Q$的\underline{無向交比}。

\section{圓交比}

\section{交比的性質}
以下性質的名稱為筆者自創，讀者於數學競賽時，可直接使用交比的性質。

如無特別說明，本章節都默認以下前提：\underline{$A$、$B$、$P$和$Q$為同一直線上的四個相異點。}
\begin{mybox}[性質1]{4.5em}{1em}{0.5em}
    有向-無向轉換性：$\left[A,B;P,Q\right]=\left|\left(A,B;P,Q\right)\right|$；
\end{mybox}
\begin{mybox}[證明1]{4.5em}{1em}{0.5em}
    由有向交比的定義可得
    \begin{equation*}
        \left|\left(A,B;P,Q\right)\right|=\left|\frac{\overrightarrow{AP}/\overrightarrow{PB}}{\overrightarrow{AQ}/\overrightarrow{QB}}\right|
        =\frac{AP/PB}{AQ/QB}=\left[A,B;P,Q\right]
    \end{equation*}
\end{mybox}
\begin{mybox}[性質2]{4.5em}{1em}{0.5em}
    有向-無向調和性：$\left(A,B;P,Q\right)=-1\Leftrightarrow\left[A,B;P,Q\right]=1$；
\end{mybox}
\begin{mybox}[證明2]{4.5em}{1em}{0.5em}
    由性質1可以直接證明性質2的充分性，下面只證明必要性：
    \begin{equation*}
        \left[A,B;P,Q\right]=1\Leftrightarrow\left(A,B;P,Q\right)=\pm 1
    \end{equation*}
    若$\left(A,B;P,Q\right)=1$，則由有向交比的定義可得
    \begin{equation*}
        \overrightarrow{AP}/\overrightarrow{PB}=\overrightarrow{AQ}/\overrightarrow{QB}
    \end{equation*}
    但對於線段$AB$來說，$P$和$Q$是兩個相異的分點，所以$\overrightarrow{AP}/\overrightarrow{PB}\neq\overrightarrow{AQ}/\overrightarrow{QB}$，矛盾。

    因此$\left(A,B;P,Q\right)$只能等於$-1$，必要性得證。
\end{mybox}
\begin{mybox}[性質3]{4.5em}{1em}{0.5em}
    有向交比的值域：$\left(A,B;P,Q\right)$的值域為$\mathbb{R}\setminus\left\{0,1\right\}$；
\end{mybox}
\begin{mybox}[證明3]{4.5em}{1em}{0.5em}
    對於$\left(A,B;P,Q\right)\neq1$的證明，已在蘊含在性質2的必要性證明當中。

    對於$\left(A,B;P,Q\right)\neq0$的證明，可直接由「$A$、$B$、$P$和$Q$四點相異」直接得出。

    對於任意實數$r\neq0,1$，我們可以任取一直線上的三個相異點$A$、$B$和$Q$，然後在同一直線取一點$P$滿足：
    $$
        \overrightarrow{BP}=-\frac{\overrightarrow{AB}}{1+r\overrightarrow{AQ}/\overrightarrow{QB}}
    $$
    此時便能滿足$\left(A,B;P,Q\right)=r$，因此有向交比的值域為$\mathbb{R}\setminus\left\{0,1\right\}$。
\end{mybox}
\begin{mybox}[性質4]{4.5em}{1em}{0.5em}
    有向交比的正值：$\left(A,B;P,Q\right)>0$當且僅當$P$和$Q$同為線段$AB$的內分點或外分點；
\end{mybox}
\begin{mybox}[證明4]{4.5em}{1em}{0.5em}
    由有向交比的定義可得
    \begin{equation*}
        \left(A,B;P,Q\right)>0\Leftrightarrow\overrightarrow{AP}/\overrightarrow{PB}
        \text{和}\overrightarrow{AQ}/\overrightarrow{QB}\text{同正或同負}\Leftrightarrow\text{$P$和$Q$同為線段$AB$的內分點或外分點}
    \end{equation*}
\end{mybox}
\begin{mybox}[性質5]{4.5em}{1em}{0.5em}
    端點-分點對交換性：$\left(A,B;P,Q\right)=\left(P,Q;A,B\right)$；
\end{mybox}
\begin{mybox}[證明5]{4.5em}{1em}{0.5em}
    由有向交比的定義可得
    \begin{equation*}
        \left(P,Q;A,B\right)=\frac{\overrightarrow{PA}/\overrightarrow{AQ}}{\overrightarrow{PB}/\overrightarrow{BQ}}
        =\frac{\overrightarrow{PA}\cdot\overrightarrow{BQ}}{\overrightarrow{PB}\cdot\overrightarrow{AQ}}=\frac{\overrightarrow{PA}/\overrightarrow{PB}}{\overrightarrow{AQ}/\overrightarrow{BQ}}=\frac{\overrightarrow{AP}/\overrightarrow{PB}}{\overrightarrow{AQ}/\overrightarrow{QB}}=\left(A,B;P,Q\right)
    \end{equation*}
\end{mybox}
\begin{mybox}[性質6]{4.5em}{1em}{0.5em}
    端點-分點倒數性：$\left(A,B;P,Q\right)=\left(A,B;Q,P\right)^{-1}=\left(B,A;P,Q\right)^{-1}$；
\end{mybox}
\begin{mybox}[證明6]{4.5em}{1em}{0.5em}
    由有向交比的定義可得
    \begin{equation*}
        \left(A,B;Q,P\right)=\frac{\overrightarrow{AQ}/\overrightarrow{QB}}{\overrightarrow{AP}/\overrightarrow{PB}}=\left(\frac{\overrightarrow{AP}/\overrightarrow{PB}}{\overrightarrow{AQ}/\overrightarrow{QB}}\right)^{-1}=\left(A,B;P,Q\right)^{-1}
    \end{equation*}
    再結合性質5可得
    \begin{equation*}
        \left(A,B;P,Q\right)=\left(P,Q;A,B\right)=\left(P,Q;B,A\right)^{-1}=\left(B,A;P,Q\right)^{-1}
    \end{equation*}
\end{mybox}
\section{交比定理}
\begin{mybox}[定理1]{4.5em}{1em}{0.5em}
    在平面內任取一點$K$，任意作四條通過$K$點的相異直線$\ell_1$、$\ell_2$、$\ell_3$和$\ell_4$。
    對於所有不通過$K$點的直線$\ell_5$，使其與直線$\ell_1$、$\ell_2$、$\ell_3$和$\ell_4$相交於$A$、$B$、$P$和$Q$四點，
    則有向分比滿足
    $$
        \left(A,B;P,Q\right)
        =\frac{\sin{\textcolor{red}{\measuredangle{\overrightarrow{AKP}}}}/\sin{\textcolor{red}{\measuredangle{\overrightarrow{PKB}}}}}
        {\sin{\textcolor{blue}{\measuredangle{\overrightarrow{AKQ}}}}/\sin{\textcolor{blue}{\measuredangle{\overrightarrow{QKB}}}}}
    $$

\end{mybox}
\begin{mybox}[證明1]{4.5em}{1em}{0.5em}
    利用有向分角定理可知
    \begin{equation*}
        \frac{\overrightarrow{AP}}{\overrightarrow{PB}}
        =\frac{\sin{\measuredangle{\overrightarrow{AKP}}}}{\sin{\measuredangle{\overrightarrow{PKB}}}}
        \cdot\frac{KA}{KB}
    \end{equation*}
    和
    \begin{equation*}
        \frac{\overrightarrow{AQ}}{\overrightarrow{QB}}
        =\frac{\sin{\measuredangle{\overrightarrow{AKQ}}}}{\sin{\measuredangle{\overrightarrow{QKB}}}}
        \cdot\frac{KA}{KB}
    \end{equation*}
    上述兩式相除可得
    \begin{equation*}
        \left(A,B;P,Q\right)=\frac{\sin{\measuredangle{\overrightarrow{AKP}}}/\sin{\measuredangle{\overrightarrow{PKB}}}}{\sin{\measuredangle{\overrightarrow{AKQ}}}/\sin{\measuredangle{\overrightarrow{QKB}}}}
    \end{equation*}
\end{mybox}

我們把上述的四條共點直線稱為「共點四線束」。觀察定理1及圖\ref{fig:cr_bundles}，
第五條直線與共點四線束相交而生成的四個交點，其有向交比與\fbox{第五條直線的相對位置\textbf{無關}}，
只與\fbox{共點四線束的夾角（有向角）\textbf{有關}}。

基於上述結論，我們可以得出定理2：
\begin{mybox}[定理2]{4.5em}{1em}{0.5em}
    在平面內任取一點$K$，任意通過$K$點的共點四線束$\ell_1$、$\ell_2$、$\ell_3$和$\ell_4$。
    任取兩條不通過$K$點的直線$\ell$和$\ell'$，兩條直線與共線四線束的交點關係如下：
    \begin{enumerate}
        \item 直線$\ell$分別與直線$\ell_1$、$\ell_2$、$\ell_3$和$\ell_4$相交於$A$、$B$、$P$和$Q$四點；
        \item 直線$\ell'$分別與直線$\ell_1$、$\ell_2$、$\ell_3$和$\ell_4$相交於$A'$、$B'$、$P'$和$Q'$四點。
    \end{enumerate}
    那麼，兩組交點點列的有向交比必定相等，即滿足
    $$
        \left(A,B;P,Q\right)=\left(A',B';P',Q'\right)
    $$

\end{mybox}

\section{調和點列}
在介紹完有向交比和無向交比的定義及性質後，我們將引入一個非常重要的概念——調和點列。
這個概念亦是數學競賽中的幾何常見題型，時常配合着角平分線或調和四邊形一起出現。

介紹調和點列之前，首先要引入另一個概念——無窮遠點：
\begin{mybox}[無窮遠點]{6em}{1em}{0.5em}
    故名思義，無窮遠點是一個存在於無窮遠處的點。
    
    對於空間上任一條直線，「無窮遠點$\infty$」皆在直線上。

    在歐氏幾何中，一對平行線永不相交，但是在射影幾何中，一對平行線是會相交於「無窮遠點$\infty$」的。

    對於任意兩點$A$、$B$，無窮遠點$\infty$的有向分割比為
    $$
        \overrightarrow{A\infty}/\overrightarrow{\infty B}=-1
    $$
    上式可以視為無窮遠點的「數值定義」。
\end{mybox}
\begin{mybox}[調和點列（定義1）]{10.5em}{1em}{0.5em}
    對於同一直線上的四個相異點$A$、$B$、$P$和$Q$，如果$\left(A,B;P,Q\right)=-1$，則稱四點$A$、$B$、$P$和$Q$為一組\underline{調和點列}。
\end{mybox}
\begin{mybox}[調和點列（定義2）]{10.5em}{1em}{0.5em}
    對於同一直線上的四個相異點$A$、$B$、$P$和$Q$，如果$\left[A,B;P,Q\right]=1$，則稱四點$A$、$B$、$P$和$Q$為一組\underline{調和點列}。
\end{mybox}
